	fontSize: number
	lineHeight: number
	letterSpacing: number
	color: `#${string}`
	backgroundColor: `#${string}`

type OptionKey = keyof SettingOptions // 'fontSize' | 'lineHeight' | 'letterSpacing' ...

class A {
	declare propertyA: string
	declare propertyB: number

	static build() {
		let a = new A()
		a.propertyA = 'default'
		a.propertyB = '100'
		return a
	}
}

const key = {
	a: 'KEY_A',

type Key = ValueOf<typeof key>
